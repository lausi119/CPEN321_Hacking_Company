  locationDifference(loc2){
    if(!global.userInfo){
        return -1;
    }
    var loc1 = global.userInfo.location;
    if(!loc1){
        return -1;
    }
    var x = Math.abs(loc1.lat-loc2.latitude);
    var y = Math.abs(loc1.long-loc2.longitude);
    y *= 111.320*Math.cos((loc2.latitude+loc1.lat)/2);
    x *= 110.574;
    return Math.sqrt(x*x+y*y);
  }

    if(distance < 0){

    else if(distance < 1){

    else{

}
